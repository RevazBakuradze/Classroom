use javatraining;

select ID, FIRST_NM,LAST_NM,ADDRESS,CITY,STATE from STUDENT;

CREATE TABLE `student` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `FIRST_NM` varchar(25) NOT NULL,
  `LAST_NM` varchar(25) DEFAULT NULL,
  `ADDRESS` varchar(25) DEFAULT NULL,
  `CITY` varchar(25) DEFAULT NULL,
  `STATE` varchar(25) DEFAULT NULL,
  `CCT_CREATETIME_DTE` datetime DEFAULT NULL,
  `CCT_UPDATETIME_DTE` datetime DEFAULT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=5173 DEFAULT CHARSET=latin1;


CREATE TABLE `course` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `TITLE` int(11) NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE KEY `UX_CMMN_COURT_CCT_TPL_ID` (`TITLE`)
) ENGINE=InnoDB AUTO_INCREMENT=5173 DEFAULT CHARSET=latin1;


CREATE TABLE `student_course` (
  `STUDENT_ID` int(11) NOT NULL,
  `COURSE_ID` int(11) NOT NULL,
  PRIMARY KEY (`STUDENT_ID`, `COURSE_ID`),
  CONSTRAINT `STUDENT_FK` FOREIGN KEY (`STUDENT_ID`) REFERENCES `student` (`ID`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `COURSE_FK` FOREIGN KEY (`COURSE_ID`) REFERENCES `course` (`ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=5173 DEFAULT CHARSET=latin1;
